            # ğŸ“Š Update progress

        return {

            "results": results,
            "summary": {
                "total_processed": len(results),
                "processing_time": time.time() - self.processing_jobs[job_id]["started_at"]

class TaskManager:
    """
    ğŸ¯ Advanced background task management system
    
    Features:
    ğŸ“Š Task queuing and prioritization
    ğŸ”„ Progress tracking
    âŒ Error handling and retries
    ğŸ“ˆ Performance monitoring
    ğŸ”” Real-time notifications
    ğŸ“Š Task analytics
    """
    
    def __init__(self):
        self.tasks: Dict[str, TaskInfo] = {}
        self.task_queues: Dict[TaskPriority, asyncio.Queue] = {
            priority: asyncio.Queue() for priority in TaskPriority
        }
        self.active_tasks: Dict[str, asyncio.Task] = {}
        self.task_functions: Dict[str, Callable] = {}
        self.max_concurrent_tasks = 10
        self.running = False
        
    def register_task_function(self, name: str, func: Callable):
        """ğŸ“‹ Register a task function"""
        self.task_functions[name] = func
        logger.info(f"ğŸ“‹ Registered task function: {name}")
    
    async def queue_task(
        self,
        name: str,
        description: str = "",
        priority: TaskPriority = TaskPriority.NORMAL,
        **kwargs
    ) -> str:
        """ğŸ“¥ Queue a new background task"""
        
        if name not in self.task_functions:
            raise ValueError(f"âŒ Unknown task function: {name}")
        
        task_id = str(uuid.uuid4())
        
        task_info = TaskInfo(
            task_id=task_id,
            name=name,
            description=description or f"Task: {name}",
            status=TaskStatus.PENDING,
            priority=priority,
            created_at=datetime.utcnow(),
            metadata=kwargs
        )
        
        self.tasks[task_id] = task_info
        await self.task_queues[priority].put((task_id, kwargs))
        
        logger.info(f"ğŸ“¥ Queued task {task_id}: {name} (priority: {priority.value})")
        
        return task_id
    
    async def start_worker_pool(self):
        """ğŸ‘· Start background worker pool"""
        self.running = True
        
        # Start workers for each priority level
        workers = []
        for priority in [TaskPriority.CRITICAL, TaskPriority.HIGH, TaskPriority.NORMAL, TaskPriority.LOW]:
            for i in range(2):  # 2 workers per priority level
                worker = asyncio.create_task(self._worker(priority, f"worker-{priority.value}-{i}"))
                workers.append(worker)
        
        logger.info(f"ğŸ‘· Started {len(workers)} background workers")
        
        try:
            await asyncio.gather(*workers)
        except asyncio.CancelledError:
            logger.info("ğŸ›‘ Background workers stopped")
    
    async def _worker(self, priority: TaskPriority, worker_name: str):
        """ğŸ‘· Individual worker process"""
        logger.info(f"ğŸ‘· Worker {worker_name} started for {priority.value} priority tasks")
        
        while self.running:
            try:
                # Get task from priority queue
                task_id, kwargs = await asyncio.wait_for(
                    self.task_queues[priority].get(),
                    timeout=1.0
                )
                
                if len(self.active_tasks) >= self.max_concurrent_tasks:
                    # Put task back in queue if too many active tasks
                    await self.task_queues[priority].put((task_id, kwargs))
                    await asyncio.sleep(0.1)
                    continue
                
                # Execute task
                task_coroutine = self._execute_task(task_id, kwargs)
                active_task = asyncio.create_task(task_coroutine)
                self.active_tasks[task_id] = active_task
                
                # Don't await here - let it run in background
                active_task.add_done_callback(
                    lambda t, tid=task_id: self.active_tasks.pop(tid, None)
                )
                
            except asyncio.TimeoutError:
                # No tasks in queue, continue
                continue
            except Exception as e:
                logger.error(f"âŒ Worker {worker_name} error: {e}")
                await asyncio.sleep(1)
    
    async def _execute_task(self, task_id: str, kwargs: Dict[str, Any]):
        """ğŸ”„ Execute individual task"""
        task_info = self.tasks.get(task_id)
        if not task_info:
            logger.error(f"âŒ Task {task_id} not found")
            return
        
        try:
            # Update task status
            task_info.status = TaskStatus.RUNNING
            task_info.started_at = datetime.utcnow()
            
            logger.info(f"ğŸ”„ Starting task {task_id}: {task_info.name}")
            
            # Get task function
            func = self.task_functions[task_info.name]
            
            # Execute task with progress callback
            progress_callback = lambda p: self.update_task_progress(task_id, p)
            
            if asyncio.iscoroutinefunction(func):
                result = await func(progress_callback=progress_callback, **kwargs)
            else:
                result = func(progress_callback=progress_callback, **kwargs)
            
            # Task completed successfully
            task_info.status = TaskStatus.COMPLETED
            task_info.completed_at = datetime.utcnow()
            task_info.result = result
            task_info.progress = 100
            
            logger.info(f"âœ… Task {task_id} completed successfully")
            
        except Exception as e:
            logger.error(f"âŒ Task {task_id} failed: {e}")
            
            # Handle task failure
            task_info.error = str(e)
            task_info.retry_count += 1
            
            if task_info.retry_count <= task_info.max_retries:
                # Retry task
                task_info.status = TaskStatus.RETRYING
                await asyncio.sleep(2 ** task_info.retry_count)  # Exponential backoff
                await self.task_queues[task_info.priority].put((task_id, kwargs))
                logger.info(f"ğŸ”„ Retrying task {task_id} (attempt {task_info.retry_count})")
            else:
                # Task failed permanently
                task_info.status = TaskStatus.FAILED
                task_info.completed_at = datetime.utcnow()
    
    def update_task_progress(self, task_id: str, progress: int):
        """ğŸ“Š Update task progress"""
        if task_id in self.tasks:
            self.tasks[task_id].progress = max(0, min(100, progress))
            logger.debug(f"ğŸ“Š Task {task_id} progress: {progress}%")
    
    def get_task_info(self, task_id: str) -> Optional[TaskInfo]:
        """ğŸ“‹ Get task information"""
        return self.tasks.get(task_id)
    
    def get_task_status(self, task_id: str) -> Optional[TaskStatus]:
        """ğŸ“Š Get task status"""
        task = self.tasks.get(task_id)
        return task.status if task else None
    
    def cancel_task(self, task_id: str) -> bool:
        """âŒ Cancel a running task"""
        if task_id in self.active_tasks:
            self.active_tasks[task_id].cancel()
            if task_id in self.tasks:
                self.tasks[task_id].status = TaskStatus.CANCELLED
            logger.info(f"âŒ Cancelled task {task_id}")
            return True
        return False
    
    def get_task_statistics(self) -> Dict[str, Any]:
        """ğŸ“Š Get comprehensive task statistics"""
        total_tasks = len(self.tasks)
        status_counts = {}
        
        for status in TaskStatus:
            status_counts[status.value] = sum(
                1 for task in self.tasks.values() 
                if task.status == status
            )
        
        active_count = len(self.active_tasks)
        queue_sizes = {
            priority.value: self.task_queues[priority].qsize()
            for priority in TaskPriority
        }
        
        return {
            "total_tasks": total_tasks,
            "active_tasks": active_count,
            "status_breakdown": status_counts,
            "queue_sizes": queue_sizes,
            "worker_pool_status": "running" if self.running else "stopped"
        }

class AdvancedConnectionManager:
    """
    ğŸ”— Advanced WebSocket connection manager
    
    Features:
    ğŸ¯ Channel-based messaging
    ğŸ” Authentication support
    ğŸ“Š Connection analytics
    ğŸ’¾ Message persistence
    ğŸ”” Automated notifications
    ğŸ“ˆ Real-time metrics
    """
    
    def __init__(self):
        # ğŸ”— Active connections
        self.connections: Dict[str, WebSocketConnection] = {}
        
        # ğŸ¯ Channel management
        self.channels: Dict[str, Set[str]] = {}  # channel -> connection_ids
        
        # ğŸ“Š Connection analytics
        self.connection_stats = {
            "total_connections": 0,
            "current_connections": 0,
            "messages_sent": 0,
            "messages_received": 0,
            "peak_connections": 0
        }
        
        # ğŸ”” Message queue for offline users
        self.offline_messages: Dict[str, List[Dict[str, Any]]] = {}
    
    async def connect(
        self,
        websocket: WebSocket,
        connection_id: str,
        user_id: str = None,
        username: str = None,
        channels: List[str] = None
    ):
        """ğŸ”— Handle new WebSocket connection"""
        
        await websocket.accept()
        
        connection = WebSocketConnection(
            websocket=websocket,
            user_id=user_id,
            username=username or f"User_{connection_id[:8]}",
            channels=set(channels or ["global"])
        )
        
        self.connections[connection_id] = connection
        
        # ğŸ“Š Add to channels
        for channel in connection.channels:
            if channel not in self.channels:
                self.channels[channel] = set()
            self.channels[channel].add(connection_id)
        
        # ğŸ“ˆ Update statistics
        self.connection_stats["total_connections"] += 1
        self.connection_stats["current_connections"] += 1
        self.connection_stats["peak_connections"] = max(
            self.connection_stats["peak_connections"],
            self.connection_stats["current_connections"]
        )
        
        # ğŸ“¢ Notify about new connection
        await self.broadcast_to_channel(
            "global",
            {
                "type": MessageType.USER_ACTIVITY,
                "message": f"ğŸ‘‹ {connection.username} joined",
                "user_id": user_id,
                "username": connection.username,
                "timestamp": datetime.utcnow().isoformat()
            },
            exclude=[connection_id]
        )
        
        # ğŸ“§ Send offline messages if any
        if user_id and user_id in self.offline_messages:
            for message in self.offline_messages[user_id]:
                await self.send_personal_message(connection_id, message)
            del self.offline_messages[user_id]
        
        logger.info(f"ğŸ”— WebSocket connected: {connection_id} ({connection.username})")
    
    async def disconnect(self, connection_id: str):
        """ğŸ”Œ Handle WebSocket disconnection"""
        
        connection = self.connections.get(connection_id)
        if not connection:
            return
        
        # ğŸ“Š Remove from channels
        for channel in connection.channels:
            if channel in self.channels:
                self.channels[channel].discard(connection_id)
                # Clean up empty channels
                if not self.channels[channel]:
                    del self.channels[channel]
        
        # ğŸ“ˆ Update statistics
        self.connection_stats["current_connections"] -= 1
        
        # ğŸ“¢ Notify about disconnection
        await self.broadcast_to_channel(
            "global",
            {
                "type": MessageType.USER_ACTIVITY,
                "message": f"ğŸ‘‹ {connection.username} left",
                "user_id": connection.user_id,
                "username": connection.username,
                "timestamp": datetime.utcnow().isoformat()
            },
            exclude=[connection_id]
        )
        
        # ğŸ—‘ï¸ Clean up connection
        del self.connections[connection_id]
        
        logger.info(f"ğŸ”Œ WebSocket disconnected: {connection_id} ({connection.username})")
    
    async def send_personal_message(self, connection_id: str, message: Dict[str, Any]):
        """ğŸ“§ Send message to specific connection"""
        
        connection = self.connections.get(connection_id)
        if not connection:
            logger.warning(f"ğŸ“§ Connection {connection_id} not found for personal message")
            return False
        
        try:
            await connection.websocket.send_text(json.dumps(message))
            self.connection_stats["messages_sent"] += 1
            return True
            
        except Exception as e:
            logger.error(f"âŒ Failed to send personal message to {connection_id}: {e}")
            # Connection might be broken, disconnect it
            await self.disconnect(connection_id)
            return False
    
    async def broadcast_to_channel(
        self,
        channel: str,
        message: Dict[str, Any],
        exclude: List[str] = None
    ):
        """ğŸ“¢ Broadcast message to all connections in a channel"""
        
        if channel not in self.channels:
            logger.warning(f"ğŸ“¢ Channel {channel} not found")
            return 0
        
        exclude = exclude or []
        sent_count = 0
        failed_connections = []
        
        # ğŸ“¡ Send to all connections in channel
        for connection_id in self.channels[channel].copy():
            if connection_id in exclude:
                continue
                
            success = await self.send_personal_message(connection_id, message)
            if success:
                sent_count += 1
            else:
                failed_connections.append(connection_id)
        
        # ğŸ§¹ Clean up failed connections
        for failed_id in failed_connections:
            await self.disconnect(failed_id)
        
        logger.debug(f"ğŸ“¢ Broadcast to {channel}: {sent_count} messages sent")
        return sent_count
    
    async def broadcast_to_all(self, message: Dict[str, Any], exclude: List[str] = None):
        """ğŸ“¢ Broadcast message to all connections"""
        return await self.broadcast_to_channel("global", message, exclude)
    
    async def send_to_user(self, user_id: str, message: Dict[str, Any]):
        """ğŸ‘¤ Send message to specific user (all their connections)"""
        
        sent_count = 0
        
        for connection_id, connection in self.connections.items():
            if connection.user_id == user_id:
                success = await self.send_personal_message(connection_id, message)
                if success:
                    sent_count += 1
        
        # ğŸ’¾ Store message for offline user if no active connections
        if sent_count == 0:
            if user_id not in self.offline_messages:
                self.offline_messages[user_id] = []
            self.offline_messages[user_id].append(message)
            
            # ğŸ§¹ Limit offline message queue size
            if len(self.offline_messages[user_id]) > 100:
                self.offline_messages[user_id] = self.offline_messages[user_id][-100:]
        
        return sent_count
    
    async def join_channel(self, connection_id: str, channel: str):
        """ğŸ”— Add connection to a channel"""
        
        connection = self.connections.get(connection_id)
        if not connection:
            return False
        
        connection.channels.add(channel)
        
        if channel not in self.channels:
            self.channels[channel] = set()
        self.channels[channel].add(connection_id)
        
        logger.info(f"ğŸ”— Connection {connection_id} joined channel {channel}")
        return True
    
    async def leave_channel(self, connection_id: str, channel: str):
        """ğŸ”Œ Remove connection from a channel"""
        
        connection = self.connections.get(connection_id)
        if not connection:
            return False
        
        connection.channels.discard(channel)
        
        if channel in self.channels:
            self.channels[channel].discard(connection_id)
            # Clean up empty channel
            if not self.channels[channel]:
                del self.channels[channel]
        
        logger.info(f"ğŸ”Œ Connection {connection_id} left channel {channel}")
        return True
    
    def get_connection_info(self, connection_id: str) -> Optional[Dict[str, Any]]:
        """â„¹ï¸ Get connection information"""
        
        connection = self.connections.get(connection_id)
        if not connection:
            return None
        
        return {
            "connection_id": connection_id,
            "user_id": connection.user_id,
            "username": connection.username,
            "channels": list(connection.channels),
            "connected_at": connection.connected_at.isoformat(),
            "last_ping": connection.last_ping.isoformat() if connection.last_ping else None,
            "metadata": connection.metadata
        }
    
    def get_statistics(self) -> Dict[str, Any]:
        """ğŸ“Š Get comprehensive connection statistics"""
        
        return {
            **self.connection_stats,
            "active_connections": list(self.connections.keys()),
            "active_channels": {
                channel: len(connections) 
                for channel, connections in self.channels.items()
            },
            "offline_message_queues": {
                user_id: len(messages)
                for user_id, messages in self.offline_messages.items()
            }
        }
