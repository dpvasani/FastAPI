    ğŸ”— Connection pooling
    ğŸ“Š Query optimization
    ğŸ”’ Transaction management

    def __init__(self):

            "DATABASE_URL",

        )
        self.min_connections = 5
        self.max_connections = 20
        self.query_timeout = 30.0
        
        logger.info("ğŸ’¾ Database service initialized")
    
    async def __call__(self) -> "DatabaseService":
        """
        ğŸ¯ Dependency injection entry point
        
        This method is called by FastAPI when the dependency is resolved.
        Ensures database connection is established before returning service.
        """

    async def connect(self):
        """ğŸ”— Establish database connection pool"""
        try:
            self.pool = await asyncpg.create_pool(
                self.connection_string,
                min_size=self.min_connections,
                max_size=self.max_connections,
                command_timeout=self.query_timeout
            )
            logger.info(f"ğŸ”— Database pool created: {self.min_connections}-{self.max_connections} connections")
            
        except Exception as e:
            logger.error(f"âŒ Database connection failed: {e}")
            raise HTTPException(
                status_code=503, 
                detail="ğŸš« Database service unavailable"
            )
    
    async def disconnect(self):
        """ğŸ”Œ Close database connection pool"""
        if self.pool:
            await self.pool.close()
            logger.info("ğŸ”Œ Database pool closed")
    
    @asynccontextmanager
    async def get_connection(self) -> AsyncGenerator[asyncpg.Connection, None]:
        """ğŸ”— Get database connection from pool"""
        if not self.pool:
            raise HTTPException(
                status_code=503,
                detail="ğŸš« Database pool not initialized"
            )
        
        async with self.pool.acquire() as connection:
            try:
                yield connection
            except Exception as e:
                logger.error(f"âŒ Database operation failed: {e}")
                raise
    
    async def execute_query(
        self, 
        query: str, 
        *args,
        fetch_one: bool = False,
        fetch_all: bool = False
    ) -> Any:
        """
        ğŸ”„ Execute database query with error handling

        ğŸ’¾ Data insertion/updates
        ğŸ” Complex queries
        ğŸ“ˆ Analytics queries

            try:
                if fetch_one:
                    result = await conn.fetchrow(query, *args)
                elif fetch_all:
                    result = await conn.fetch(query, *args)
                else:
                    result = await conn.execute(query, *args)
                
                execution_time = time.time() - start_time
                logger.info(f"ğŸ“Š Query executed in {execution_time:.4f}s")
                
                return result
                
            except Exception as e:
                execution_time = time.time() - start_time
                logger.error(f"âŒ Query failed after {execution_time:.4f}s: {e}")
                raise HTTPException(
                    status_code=500,
                    detail="ğŸš« Database query failed"
                )

            for session_id in sessions:
                session_data = self.redis_client.get(f"session:{session_id}")

                    session_details.append((session_id, data["created_at"]))

            session_details.sort(key=lambda x: x[1])
            sessions_to_remove = session_details[:-self.max_sessions_per_user]

                self.revoke_session(session_id)
                logger.info(f"ğŸ§¹ Removed old session: {session_id}")

    async def check_availability(self, items: List[Dict[str, Any]]) -> Dict[str, Any]:
        """ğŸ“¦ Check item availability"""
        
        availability_results = []
        all_available = True
        
        for item in items:
            product_id = item["product_id"]
            requested_quantity = item["quantity"]
            
            # ğŸ“Š Get current stock
            stock_info = await self.db.execute_query(
                "SELECT quantity, reserved_quantity FROM inventory WHERE product_id = $1",
                product_id,
                fetch_one=True
            )
            
            if stock_info:
                available_quantity = stock_info["quantity"] - stock_info["reserved_quantity"]
                is_available = available_quantity >= requested_quantity
                
                availability_results.append({
                    "product_id": product_id,
                    "requested_quantity": requested_quantity,
                    "available_quantity": available_quantity,
                    "is_available": is_available
                })
                
                if not is_available:
                    all_available = False
            else:
                # Product not found in inventory
                availability_results.append({
                    "product_id": product_id,
                    "requested_quantity": requested_quantity,
                    "available_quantity": 0,
                    "is_available": False
                })
                all_available = False
        
        return {
            "all_available": all_available,
            "items": availability_results
        }
    
    async def reserve_items(self, items: List[Dict[str, Any]]) -> bool:
        """ğŸ“¦ Reserve items for order"""
        
        try:
            # ğŸ”„ Start database transaction

                async with conn.transaction():
                    for item in items:
                        await conn.execute(
                            """
                            UPDATE inventory 
                            SET reserved_quantity = reserved_quantity + $1
                            WHERE product_id = $2
                            """,
                            item["quantity"],
                            item["product_id"]
                        )
            
            logger.info(f"ğŸ“¦ Reserved {len(items)} items")
            return True
            
        except Exception as e:
            logger.error(f"âŒ Failed to reserve items: {e}")
            return False

async def global_rate_limiter(request: Request) -> bool:
    """
    ğŸš¦ Global rate limiting for all endpoints
    
    Applied to: ALL endpoints in the application
    Use Cases:
    ğŸ›¡ï¸ DDoS protection
    ğŸš« API abuse prevention
    ğŸ“Š Resource management
    âš¡ Performance optimization
    """
    
    client_ip = request.client.host if request.client else "unknown"
    
    # ğŸ”‘ Create rate limit key
    rate_limit_key = f"rate_limit:{client_ip}"
    
    # ğŸ“Š Check current request count (using Redis or in-memory cache)
    try:
        # Using Redis for distributed rate limiting
        import redis
        redis_client = redis.Redis(host="localhost", port=6379, db=1)
        
        current_requests = redis_client.get(rate_limit_key)
        if current_requests is None:
            # ğŸ†• First request from this IP
            redis_client.setex(rate_limit_key, 60, 1)  # 1 request in 60 seconds window
            return True
        
        current_count = int(current_requests)
        max_requests = 100  # 100 requests per minute
        
        if current_count >= max_requests:
            logger.warning(f"ğŸš¦ Rate limit exceeded for IP: {client_ip}")
            raise HTTPException(
                status_code=429,
                detail="ğŸš¦ Too many requests. Please try again later.",
                headers={"Retry-After": "60"}
            )
        
        # ğŸ“ˆ Increment request count
        redis_client.incr(rate_limit_key)
        
        logger.debug(f"ğŸš¦ Rate limit check passed: {client_ip} ({current_count + 1}/{max_requests})")
        return True
        
    except redis.RedisError:
        # ğŸš¨ If Redis is down, allow requests but log warning
        logger.warning("ğŸš¨ Rate limiting service unavailable - allowing request")
        return True

async def global_security_validator(request: Request) -> Dict[str, Any]:
    """
    ğŸ›¡ï¸ Global security validation for all requests
    
    Applied to: ALL endpoints in the application
    Use Cases:
    ğŸš« Malicious request detection
    ğŸ”’ Input sanitization
    ğŸ›¡ï¸ Header validation
    ğŸ“Š Security analytics
    """
    
    security_info = {
        "is_secure": True,
        "warnings": [],
        "blocked": False
    }
    
    # ğŸ” Check for suspicious patterns in URL
    suspicious_patterns = [
        "../", "..\\",  # Directory traversal
        "<script", "</script>",  # XSS attempts
        "union select", "drop table",  # SQL injection
        "eval(", "exec(",  # Code injection
    ]
    
    url_path = str(request.url.path).lower()
    query_string = str(request.url.query).lower()
    
    for pattern in suspicious_patterns:
        if pattern in url_path or pattern in query_string:
            security_info["warnings"].append(f"Suspicious pattern detected: {pattern}")
            security_info["is_secure"] = False
            
            logger.warning(f"ğŸš« Suspicious request from {request.client.host}: {pattern}")
    
    # ğŸ” Validate headers
    user_agent = request.headers.get("user-agent", "")
    if len(user_agent) > 1000:  # Unusually long user agent
        security_info["warnings"].append("Unusually long User-Agent header")
    
    # ğŸš« Block severely suspicious requests
    if len(security_info["warnings"]) >= 3:
        security_info["blocked"] = True
        raise HTTPException(
            status_code=400,
            detail="ğŸš« Request blocked due to security concerns"
        )
    
    # ğŸ—‚ï¸ Store security info in request state
    request.state.security_info = security_info
    
    return security_info
